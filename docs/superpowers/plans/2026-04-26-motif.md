**Goal:** Build `motif`, a pure Erlang library that extracts ranked keyword/phrase candidates from text using the RAKE algorithm with built-in stop-word lists for French, English, and German.

**Architecture:** Four modules: `motif` (public API), `motif_rake` (RAKE scoring), `motif_stopwords` (bundled stop-word lists), `motif_lang` (language auto-detection). No external dependencies — pure Erlang/OTP.

```erlang
-module(motif_lang).
-export([detect/1]).

%% Detect the language of Text by counting stop-word hits.
%% Samples the first 200 words; falls back to en on a tie.
-spec detect(binary()) -> fr | en | de.
detect(Text) ->
    Words  = sample_words(Text, 200),
    Counts = [{Lang, count_hits(Words, motif_stopwords:get(Lang))}
              || Lang <- [fr, en, de]],
    {Best, _} = lists:foldl(
        fun({L, C}, {_BL, BC}) when C > BC -> {L, C};
           (_,      Acc)                   -> Acc
        end,
        {en, 0}, Counts),
    Best.

%% Lowercase, split on whitespace/punctuation, take first N non-empty words.
sample_words(Text, N) ->
    Lower = string:lowercase(Text),
    Parts = re:split(Lower, <<"[^a-zàâäéèêëîïôùûüÿœçæäöüß]+">>,
                     [{return, binary}, trim]),
    Words = [W || W <- Parts, W =/= <<>>],
    lists:sublist(Words, N).

count_hits(Words, StopWords) ->
    Set = sets:from_list(StopWords),
    length([W || W <- Words, sets:is_element(W, Set)]).
```

```erlang
-module(motif_rake).
-export([extract/2]).

%% Extract keyword candidates from Text using the RAKE algorithm.
%% StopWords is a list of lowercase binary stop words.
%% Returns [{Candidate, Score}] sorted by score descending, deduplicated.
-spec extract(binary(), [binary()]) -> [{binary(), float()}].
extract(Text, StopWords) ->
    StopSet    = sets:from_list(StopWords),
    Sentences  = split_sentences(Text),
    Candidates = lists:flatmap(fun(S) -> phrase_candidates(S, StopSet) end, Sentences),
    case Candidates of
        [] -> [];
        _  ->
            Scores = word_scores(Candidates),
            Ranked = rank_candidates(Candidates, Scores),
            deduplicate(Ranked)
    end.

%% Split text into sentences on . ! ? and newlines.
split_sentences(Text) ->
    Lower = string:lowercase(Text),
    Parts = re:split(Lower, <<"[.!?\\n]+">>, [{return, binary}, trim]),
    [string:trim(P) || P <- Parts, string:trim(P) =/= <<>>].

%% Split a sentence into candidate phrases by cutting on stop words.
%% Single-character words are also treated as separators.
phrase_candidates(Sentence, StopSet) ->
    Words   = tokenize(Sentence),
    Phrases = split_on_stops(Words, StopSet, [], []),
    [P || P <- Phrases, length(P) > 0].

tokenize(Text) ->
    Parts = re:split(Text, <<"\\s+">>, [{return, binary}, trim]),
    Clean = [clean_word(W) || W <- Parts],
    [W || W <- Clean, W =/= <<>>].

%% Strip leading/trailing non-alpha characters; keep hyphens inside words.
clean_word(Word) ->
    Step1 = re:replace(Word, <<"^[^a-zàâäéèêëîïôùûüÿœçæäöüß-]+">>,
                       <<"">>, [{return, binary}]),
    re:replace(Step1, <<"[^a-zàâäéèêëîïôùûüÿœçæäöüß-]+$">>,
               <<"">>, [{return, binary}]).

split_on_stops([], _StopSet, Current, Acc) ->
    emit(Current, Acc);
split_on_stops([W | Rest], StopSet, Current, Acc) ->
    IsStop = sets:is_element(W, StopSet) orelse byte_size(W) =< 1,
    case IsStop of
        true  -> split_on_stops(Rest, StopSet, [], emit(Current, Acc));
        false -> split_on_stops(Rest, StopSet, Current ++ [W], Acc)
    end.

emit([], Acc)     -> Acc;
emit(Phrase, Acc) -> [Phrase | Acc].

%% Compute RAKE word scores: deg(w) / freq(w).
word_scores(Candidates) ->
    Freq   = word_freq(Candidates),
    Degree = word_degree(Candidates),
    maps:map(fun(Word, Deg) ->
        Deg / max(1, maps:get(Word, Freq, 1))
    end, Degree).

word_freq(Candidates) ->
    lists:foldl(fun(Phrase, Acc) ->
        lists:foldl(fun(W, A) -> A#{W => maps:get(W, A, 0) + 1} end, Acc, Phrase)
    end, #{}, Candidates).

%% deg(w) = number of times w co-occurs with other words in any candidate phrase.
%% For a phrase of length L, each word gets degree L (itself + L-1 co-occurrences).
word_degree(Candidates) ->
    lists:foldl(fun(Phrase, Acc) ->
        Deg = length(Phrase),
        lists:foldl(fun(W, A) -> A#{W => maps:get(W, A, 0) + Deg} end, Acc, Phrase)
    end, #{}, Candidates).

rank_candidates(Candidates, WordScores) ->
    Scored = [{iolist_to_binary(lists:join(<<" ">>, Phrase)),
               lists:sum([maps:get(W, WordScores, 0.0) || W <- Phrase])}
              || Phrase <- Candidates],
    lists:sort(fun({_, A}, {_, B}) -> A >= B end, Scored).

deduplicate(Candidates) ->
    {Result, _Seen} = lists:foldl(
        fun({C, S}, {Acc, Seen}) ->
            case sets:is_element(C, Seen) of
                true  -> {Acc, Seen};
                false -> {[{C, S} | Acc], sets:add_element(C, Seen)}
            end
        end,
        {[], sets:new()}, Candidates),
    lists:reverse(Result).
```

```erlang
-module(motif).
-export([extract/1, extract/2, stop_words/1]).

%% Extract keyword candidates with default options (lang => auto, no max limit).
-spec extract(binary()) -> [{binary(), float()}].
extract(Text) ->
    extract(Text, #{}).

%% Extract keyword candidates.
%% Options:
%%   max  => pos_integer()        cap the number of results (default: unlimited)
%%   lang => fr | en | de | auto  stop-word language (default: auto)
-spec extract(binary(), #{max => pos_integer(), lang => fr | en | de | auto}) ->
    [{binary(), float()}].
extract(Text, Opts) ->
    Lang       = resolve_lang(maps:get(lang, Opts, auto), Text),
    Stops      = motif_stopwords:get(Lang),
    Candidates = motif_rake:extract(Text, Stops),
    case maps:get(max, Opts, undefined) of
        undefined -> Candidates;
        Max       -> lists:sublist(Candidates, Max)
    end.

%% Return the built-in stop-word list for a language.
-spec stop_words(fr | en | de) -> [binary()].
stop_words(Lang) ->
    motif_stopwords:get(Lang).

resolve_lang(auto, Text) -> motif_lang:detect(Text);
resolve_lang(Lang, _Text) -> Lang.
```

````markdown
# motif

Pure Erlang keyword and topic extraction using the
[RAKE](https://www.researchgate.net/publication/227988510) algorithm.
Supports French, English, and German with built-in stop-word lists.
No external dependencies.

## Installation

```erlang
%% rebar.config
{deps, [{motif, "0.1.0"}]}.
```

## Quick start

```erlang
%% Extract from English text
Results = motif:extract(<<"Red roses are a symbol of love and beauty.">>),
%% [{<<"red roses">>, 4.0}, {<<"love">>, 1.0}, {<<"beauty">>, 1.0}]

%% Explicit language + max results
Top3 = motif:extract(Text, #{lang => fr, max => 3}),

%% Auto-detect language (samples first 200 words)
Auto = motif:extract(Text, #{lang => auto}),

%% Get the stop-word list for a language
Stops = motif:stop_words(fr).
```

## API

```erlang
%% Extract keyword candidates. Returns [{Keyword, Score}] sorted by score desc.
-spec extract(binary()) -> [{binary(), float()}].
-spec extract(binary(), #{max  => pos_integer(),
                           lang => fr | en | de | auto}) -> [{binary(), float()}].

%% Return the built-in stop-word list for a language.
-spec stop_words(fr | en | de) -> [binary()].
```

## Algorithm

RAKE (Rapid Automatic Keyword Extraction):

1. Split text into sentences on `. ! ?`
2. Within each sentence, split into candidate phrases on stop words
3. Score each word: `degree(word) / frequency(word)`
   where `degree(w)` = sum of phrase lengths containing `w`
4. Score each candidate: sum of its word scores
5. Return sorted by score descending, deduplicated

Multi-word phrases with co-occurring rare words score highest.

## Language detection

`lang => auto` samples the first 200 words, counts stop-word hits per
language, and picks the language with the most hits. Falls back to `en`
on a tie or empty input.
````
