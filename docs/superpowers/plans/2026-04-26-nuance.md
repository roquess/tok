# nuance Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `nuance_erl`, an escript that replicates the Python `nuance/topics.py` CLI exactly — reads a text file, extracts topic candidates with `motif`, scores them with `nli`, and prints the top-K results in the same three output formats.

**Architecture:** Single `nuance.erl` escript. Pipeline: parse args → read file → `motif:extract/2` (up to 30 candidates) → `nli:score/3` per candidate → sort desc → take top-K → format and print. Model paths come from `priv/` by default or from `NLI_TOKENIZER` / `NLI_MODEL` env vars.

**Tech Stack:** Erlang/OTP, rebar3 escript, motif 0.1+, nli 0.1+, Apache-2.0.

```erlang
-module(nuance).
-export([main/1, parse_args/1]).

%% escript entry point.
main(Args) ->
    case parse_args(Args) of
        {error, Msg} ->
            io:format(standard_error, "Error: ~s~n~n~s~n", [Msg, usage()]),
            halt(1);
        {ok, Opts} ->
            run(Opts)
    end.

%% Parse CLI arguments into an options map.
%% Returns {ok, #{file, max, lang, format, quiet}} or {error, Reason}.
-spec parse_args([string()]) ->
    {ok, #{file   := string(),
           max    := pos_integer(),
           lang   := fr | en | de | auto,
           format := simple | json | verbose,
           quiet  := boolean()}} | {error, string()}.
parse_args([]) ->
    {error, "No input file specified"};
parse_args([File | Rest]) ->
    Defaults = #{file => File, max => 6, lang => auto,
                 format => verbose, quiet => false},
    parse_flags(Rest, Defaults).

parse_flags([], Opts) -> {ok, Opts};
parse_flags(["--max", N | Rest], Opts) ->
    case string:to_integer(N) of
        {Max, []} when Max > 0 -> parse_flags(Rest, Opts#{max => Max});
        _                       -> {error, "--max must be a positive integer"}
    end;
parse_flags(["--lang", L | Rest], Opts) ->
    case L of
        "fr"   -> parse_flags(Rest, Opts#{lang => fr});
        "en"   -> parse_flags(Rest, Opts#{lang => en});
        "de"   -> parse_flags(Rest, Opts#{lang => de});
        "auto" -> parse_flags(Rest, Opts#{lang => auto});
        _      -> {error, "--lang must be fr, en, de, or auto"}
    end;
parse_flags(["--format", F | Rest], Opts) ->
    case F of
        "simple"  -> parse_flags(Rest, Opts#{format => simple});
        "json"    -> parse_flags(Rest, Opts#{format => json});
        "verbose" -> parse_flags(Rest, Opts#{format => verbose});
        _         -> {error, "--format must be simple, json, or verbose"}
    end;
parse_flags(["--quiet" | Rest], Opts) ->
    parse_flags(Rest, Opts#{quiet => true});
parse_flags([Flag | _], _Opts) ->
    {error, "Unknown flag: " ++ Flag}.

usage() ->
    "Usage: nuance <file> [--max N] [--lang fr|en|de|auto] "
    "[--format simple|json|verbose] [--quiet]".

run(_Opts) ->
    io:format("not implemented~n").
```

```erlang
%% Format a list of topic binaries for output.
%% simple  — one topic per line (for piping)
%% json    — JSON array
%% verbose — numbered list
-spec format_topics([binary()], simple | json | verbose) -> binary().
format_topics(Topics, simple) ->
    iolist_to_binary([[T, "\n"] || T <- Topics]);
format_topics(Topics, json) ->
    Encoded = lists:join(",", [["\"", T, "\""] || T <- Topics]),
    iolist_to_binary(["[", Encoded, "]"]);
format_topics(Topics, verbose) ->
    Numbered = lists:zipwith(
        fun(N, T) -> [integer_to_list(N), ". ", T, "\n"] end,
        lists:seq(1, length(Topics)), Topics),
    iolist_to_binary(Numbered).
```

```erlang
%% Run the full pipeline. Exported for testing.
%% Returns {ok, [binary()]} on success, {error, term()} on failure.
-spec run_pipeline(map()) -> {ok, [binary()]} | {error, term()}.
run_pipeline(Opts) ->
    FilePath  = maps:get(file, Opts),
    Max       = maps:get(max, Opts, 6),
    Lang      = maps:get(lang, Opts, auto),
    Quiet     = maps:get(quiet, Opts, false),
    TokPath   = maps:get(tok_path,   Opts, tok_path()),
    ModelPath = maps:get(model_path, Opts, model_path()),
    case file:read_file(FilePath) of
        {error, Reason} ->
            {error, {file_error, Reason}};
        {ok, Text} ->
            maybe_log(Quiet, "Step 1: Extracting keyword candidates...~n"),
            Candidates = motif:extract(Text, #{lang => Lang, max => 30}),
            maybe_log(Quiet, "  ~p candidates extracted~n", [length(Candidates)]),
            case nli:load(TokPath, ModelPath) of
                {error, Reason} ->
                    {error, {nli_load_error, Reason}};
                {ok, NLI} ->
                    maybe_log(Quiet, "Step 2: Scoring candidates with NLI...~n"),
                    Scored  = score_all(NLI, Text, Candidates, Quiet),
                    nli:unload(NLI),
                    Sorted  = lists:sort(fun({_, A}, {_, B}) -> A >= B end, Scored),
                    TopK    = lists:sublist(Sorted, Max),
                    Topics  = [C || {C, _} <- TopK],
                    {ok, Topics}
            end
    end.

score_all(NLI, Text, Candidates, Quiet) ->
    Total = length(Candidates),
    {Scored, _} = lists:foldl(
        fun({C, _RakeScore}, {Acc, Idx}) ->
            maybe_log(Quiet, "  [~p/~p] ~s~n", [Idx, Total, C]),
            Score = case nli:score(NLI, Text, C) of
                {ok, S}     -> S;
                {error, _}  -> 0.0
            end,
            {[{C, Score} | Acc], Idx + 1}
        end,
        {[], 1}, Candidates),
    Scored.

%% run/1 used by main/1 — prints output then halts.
run(Opts) ->
    Format = maps:get(format, Opts),
    Quiet  = maps:get(quiet, Opts),
    case run_pipeline(Opts) of
        {error, {file_error, Reason}} ->
            io:format(standard_error, "Error reading file: ~p~n", [Reason]),
            halt(1);
        {error, {nli_load_error, Reason}} ->
            io:format(standard_error, "Error loading NLI model: ~p~n", [Reason]),
            halt(1);
        {error, Reason} ->
            io:format(standard_error, "Error: ~p~n", [Reason]),
            halt(1);
        {ok, Topics} ->
            maybe_log(Quiet, "~n"),
            Output = format_topics(Topics, Format),
            io:format("~s", [Output])
    end.

%% Default model paths — override with NLI_TOKENIZER / NLI_MODEL env vars.
tok_path() ->
    case os:getenv("NLI_TOKENIZER") of
        false -> filename:join(priv_dir(), "tokenizer.json");
        Path  -> Path
    end.

model_path() ->
    case os:getenv("NLI_MODEL") of
        false -> filename:join(priv_dir(), "model.onnx");
        Path  -> Path
    end.

priv_dir() ->
    case code:priv_dir(nuance) of
        {error, _} -> "priv";
        Dir        -> Dir
    end.

maybe_log(true,  _Fmt)         -> ok;
maybe_log(false, Fmt)           -> io:format(standard_error, Fmt, []).

maybe_log(true,  _Fmt, _Args)  -> ok;
maybe_log(false, Fmt,  Args)   -> io:format(standard_error, Fmt, Args).
```

````markdown
# nuance_erl

Erlang replacement for `nuance/topics.py` — extracts the main topics from a text
file using [motif](https://hex.pm/packages/motif) (RAKE keyword extraction) and
[nli](https://hex.pm/packages/nli) (multilingual NLI scoring).

Same CLI interface as the original Python tool. No Python at runtime.

## Getting a model

```bash
pip install optimum
optimum-cli export onnx \
    --model MoritzLaurer/mDeBERTa-v3-base-mnli-xnli \
    priv/
```

## Build

```bash
rebar3 escriptize
```

## Usage

```bash
./nuance <file> [--max N] [--lang fr|en|de|auto] [--format simple|json|verbose] [--quiet]
```

| Flag       | Default  | Description                              |
|------------|----------|------------------------------------------|
| `--max`    | 6        | Number of topics to return               |
| `--lang`   | auto     | Stop-word language for RAKE extraction   |
| `--format` | verbose  | Output format                            |
| `--quiet`  | false    | Suppress progress messages to stderr     |

## Examples

```bash
# Default (verbose output, auto-detect language)
./nuance document.txt

# Simple output for piping
./nuance document.txt --format simple --quiet | head -3

# JSON output, French, top 3
./nuance document.txt --lang fr --max 3 --format json
```

## Model path override

Set environment variables to point to alternative model files:

```bash
export NLI_TOKENIZER=/path/to/tokenizer.json
export NLI_MODEL=/path/to/model.onnx
./nuance document.txt
```

## Pipeline

```
read file
    │
    ▼
motif:extract(Text, #{max => 30, lang => Lang})
    │  up to 30 keyword candidates via RAKE
    ▼
nli:score(NLI, Text, Candidate) for each candidate
    │  entailment probability per candidate
    ▼
sort desc, take top --max
    │
    ▼
format and print to stdout
```

## License

Apache 2.0 — see [LICENSE](LICENSE).
````
